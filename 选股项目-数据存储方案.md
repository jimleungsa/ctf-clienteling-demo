## 2. 数据校验机制

### 2.1 完整性校验
```python
def validate_completeness(code, date):
    """检查数据完整性"""
    required_fields = ['open', 'high', 'low', 'close', 'volume']
    
    # 检查字段是否存在
    for field in required_fields:
        if data[field] is None or data[field] == 0:
            return False, f"字段缺失: {field}"
    
    # 检查价格逻辑
    if not (low <= open <= high and low <= close <= high):
        return False, "价格逻辑错误"
    
    # 检查成交量
    if volume <= 0:
        return False, "成交量异常"
    
    return True, "通过"
```

### 2.2 准确性校验
```python
def validate_accuracy(code, date):
    """检查数据准确性"""
    # 涨跌幅合理性检查
    change_pct = (close - prev_close) / prev_close * 100
    if abs(change_pct) > 20:  # A股涨跌停限制
        return False, f"涨跌幅异常: {change_pct}%"
    
    # 价格范围检查
    if close > 10000 or close < 0.01:
        return False, f"价格异常: {close}"
    
    # 多源对比（Baostock vs AkShare）
    baostock_data = fetch_baostock(code, date)
    akshare_data = fetch_akshare(code, date)
    
    if abs(baostock_data.close - akshare_data.close) > 0.01:
        return False, "多源数据不一致"
    
    return True, "通过"
```

## 3. 同步失败补偿机制

### 3.1 重试机制
```python
import time
from functools import wraps

def retry(max_attempts=3, delay=60):
    """重试装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts - 1:
                        raise
                    time.sleep(delay * (attempt + 1))
            return None
        return wrapper
    return decorator

@retry(max_attempts=3, delay=60)
def sync_daily_data(code, date):
    """同步日线数据（带重试）"""
    data = fetch_baostock(code, date)
    save_to_db(data)
    return True
```

def compensation_worker():
    """补偿工作线程"""
    while True:
        # 获取待补偿任务
        tasks = get_compensation_tasks()
        
        for task in tasks:
            code, date = task['code'], task['date']
            
            # 尝试补偿
            try:
                # 切换数据源

                    data = fetch_akshare(code, date)
                else:
                    data = fetch_baostock(code, date)
                
                save_to_db(data)
                mark_compensated(task['id'])
                
            except Exception as e:
                # 补偿失败，等待下次
                update_retry_count(task['id'])
        
        time.sleep(300)  # 5分钟检查一次

## 4. 每日自动同步脚本

### 4.1 主同步脚本
```python
#!/usr/bin/env python3
# sync_daily_data.py

import schedule
import time
from datetime import datetime, timedelta

def daily_sync_job():
    """每日同步任务"""
    today = datetime.now().date()
    
    # 获取所有股票代码
    stock_list = get_all_stocks()
    
    # 同步每只股票
    for code in stock_list:
        try:
            # 检查是否已同步
            if is_synced(code, today):
                continue
            
            # 同步数据
            sync_daily_data(code, today)
            
            # 校验数据
            validate_data(code, today)
            
        except Exception as e:
            alert_sync_failure(code, today, str(e))
    
    # 记录同步日志
    log_sync_summary(today)

# 每日20:30执行（收盘后）
schedule.every().day.at("20:30").do(daily_sync_job)

while True:
    schedule.run_pending()
    time.sleep(60)
```
